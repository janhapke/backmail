```typescript
// Helper to discover and validate the repository root (per D-07, D-05)
function getRepoRoot(): string {
  const opts = program.opts() as { workdir?: string }
  const startDir = opts.workdir ? path.resolve(opts.workdir) : process.cwd()
  const repoRoot = findRepository(startDir)
  if (!repoRoot) {
    // D-06: exact two-line error, no CWD path, exit 1
    console.error(
      'Error: Not inside a backmail repository.\n' +
      'Use `backmail init` to create one, or `--workdir <path>` to specify a path.'
    )
    process.exit(1)
  }
  return repoRoot
}

// Helper to load repository config using discovered root
function getConfig(): RepositoryConfig {
  try {
    const repoRoot = getRepoRoot()
    return loadRepositoryConfig(repoRoot)
  } catch (err) {
    console.error((err as Error).message)
    process.exit(1)
  }
}
```

Note: `getRepoRoot()` calls `process.exit(1)` directly (no throw) when repo not found, so the return type annotation `string` is safe. `getConfig()` wraps `loadRepositoryConfig` errors.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| CLI args → getRepoRoot() | --workdir value is user-supplied; path.resolve() normalizes it before use |
| repoRoot → loadRepositoryConfig | Config JSON read from disk; validated by Zod schema in config.ts |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07-04 | Tampering | --workdir path traversal | accept | path.resolve() normalizes but does not restrict; attacker controls their own filesystem — no security boundary crossed |
| T-07-05 | Information Disclosure | DISC-03 error message | mitigate | D-06 mandates no CWD path in error output — implemented as exact two-line string with no dynamic content |
| T-07-06 | Elevation of Privilege | Symlink in .backmail/ marker | accept | Detection uses fs.existsSync which follows symlinks; attacker would need write access to create symlink — already has filesystem access |
</threat_model>
