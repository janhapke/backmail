```typescript
// ── Phase 9: init command ────────────────────────────────────────────────────
program
  .command('init [path]')
  .description('Create a new backmail repository')
  .option('--host <host>', 'IMAP server hostname')
  .option('--port <port>', 'IMAP server port (default: 993)')
  .option('--username <username>', 'IMAP account username')
  .option('--tls', 'use TLS (default: true)')
  .option('--no-tls', 'disable TLS')
  .option('--password <password>', 'IMAP password (written to OS keyring)')
  .option(
    '--password-ref <ref>',
    'passwordRef string written directly to config (e.g. env:BACKMAIL_PASSWORD) — use this in CI environments instead of --password',
  )
  .action(async (dirPath: string | undefined, opts: {
    host?: string
    port?: string
    username?: string
    tls?: boolean
    password?: string
    passwordRef?: string
  }) => {
    const targetDir = dirPath ? path.resolve(dirPath) : process.cwd()
    const isTTY = process.stdin.isTTY !== false  // D-10: undefined treated as interactive

    // Collect host — prompt in TTY, error in non-TTY (REPO-05)
    let host: string
    if (opts.host !== undefined) {
      host = opts.host
    } else if (isTTY) {
      host = await input({ message: 'IMAP host:', required: true })
    } else {
      console.error('Error: --host is required in non-TTY mode')
      process.exit(1)
    }

    // Collect port — prompt in TTY, error in non-TTY
    let port: number
    if (opts.port !== undefined) {
      const parsed = parseInt(opts.port, 10)
      if (!Number.isInteger(parsed) || parsed < 1 || parsed > 65535) {
        console.error('Error: --port must be an integer between 1 and 65535')
        process.exit(1)
      }
      port = parsed
    } else if (isTTY) {
      const portStr = await input({
        message: 'IMAP port:',
        default: '993',
        validate: (v) => {
          const n = parseInt(v, 10)
          return (Number.isInteger(n) && n >= 1 && n <= 65535) || 'Port must be an integer 1–65535'
        },
      })
      port = parseInt(portStr, 10)
    } else {
      console.error('Error: --port is required in non-TTY mode')
      process.exit(1)
    }

    // Collect username — prompt in TTY, error in non-TTY
    let username: string
    if (opts.username !== undefined) {
      username = opts.username
    } else if (isTTY) {
      username = await input({ message: 'IMAP username:', required: true })
    } else {
      console.error('Error: --username is required in non-TTY mode')
      process.exit(1)
    }

    // Collect TLS — prompt in TTY, error in non-TTY
    // Commander --tls / --no-tls: opts.tls is true when --tls passed, false when --no-tls passed,
    // undefined when neither is passed (verified against Pitfall 4: no explicit default set).
    let tls: boolean
    if (opts.tls !== undefined) {
      tls = opts.tls
    } else if (isTTY) {
      tls = await confirm({ message: 'Use TLS?', default: true })
    } else {
      console.error('Error: --tls or --no-tls is required in non-TTY mode')
      process.exit(1)
    }

    // Collect credential — D-04: --password writes to keyring, --password-ref skips keyring
    let passwordRef: string
    if (opts.passwordRef !== undefined) {
      // --password-ref path: write ref directly, no keyring interaction (D-04)
      passwordRef = opts.passwordRef
    } else {
      // --password path or TTY prompt
      let plaintext: string
      if (opts.password !== undefined) {
        plaintext = opts.password
      } else if (isTTY) {
        plaintext = await promptPassword({ message: 'IMAP password:', mask: true })
      } else {
        console.error('Error: --password or --password-ref is required in non-TTY mode')
        process.exit(1)
      }
      // Write to keyring (D-05: fail with clear error on keyring write failure)
      try {
        new Entry('backmail', username).setPassword(plaintext)
      } catch (err) {
        console.error(
          `Error: Failed to write password to OS keyring: ${getErrorMessage(err)}\n` +
          'Use --password-ref env:BACKMAIL_PASSWORD for CI environments.',
        )
        process.exit(1)
      }
      passwordRef = `keyring:service=backmail;account=${username}`
    }

    // Build config and call core (D-09)
    const config: RepositoryConfig = { host, port, username, tls, passwordRef }

    try {
      await initRepository(targetDir, config, passwordRef)
      console.log(`Initialized backmail repository at ${targetDir}`)
    } catch (err) {
      console.error(getErrorMessage(err))
      process.exit(1)
    }
  })
```

IMPORTANT NOTES:
- The `getErrorMessage` helper already exists in the file — do NOT redeclare it. Use it directly.
- `RepositoryConfig` is already imported via `import type { RepositoryConfig } from '../core/index.js'` — do NOT re-import.
- `path` is already imported — do NOT re-import.
- `program.parse(process.argv)` must remain the LAST line of the file after this block.
- The `isTTY` check uses `process.stdin.isTTY !== false` not `=== true` — this treats `undefined` (no pipe, no TTY flag) as interactive mode, per D-10.
- For the `--tls`/`--no-tls` pair: Commander does NOT set a default for this pair unless explicitly provided via `.option('--tls', '...', true)`. The current code uses `opts.tls !== undefined` to detect "neither flag passed" — requiring the user to explicitly choose in non-TTY, and defaulting to the confirm prompt in TTY. This avoids Pitfall 4.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| user input → CLI flags | host, port, username come from CLI flags or TTY prompts |
| --password flag → OS keyring | Plaintext password in process args (appears in ps aux) |
| CLI → initRepository() | targetDir resolved to absolute path before passing to core |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-9-03-01 | Information Disclosure | --password flag in process args | accept | Document --password-ref env:BACKMAIL_PASSWORD as preferred for CI; password in flags appears in `ps aux`. Help text explicitly mentions env: syntax (D-04, --password-ref option description). |
| T-9-03-02 | Tampering | path positional arg | mitigate | `path.resolve(dirPath)` collapses ../ sequences and makes path absolute before use |
| T-9-03-03 | Information Disclosure | keyring write failure error message | mitigate | Error message uses getErrorMessage(err) which does not include the plaintext password; error comes from OS keyring layer |
| T-9-03-04 | Denial of Service | non-TTY without flags | mitigate | Explicit error + exit(1) for each missing required param in non-TTY mode (REPO-05) |
| T-9-03-05 | Elevation of Privilege | prompts in non-TTY | mitigate | isTTY check before EVERY prompt call (Pitfall 1 mitigation); inquirer would throw otherwise |
</threat_model>
